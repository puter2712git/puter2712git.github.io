## 시도

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
	int n_count, m_count;
	bool is_included = false;

	cin >> n_count;
	vector<long long int> n_array(n_count);
	for (int i = 0; i < n_count; i++) {
		cin >> n_array[i];
	}

	cin >> m_count;
	vector<long long int> m_array(m_count);
	for (int i = 0; i < m_count; i++) {
		cin >> m_array[i];
	}

	for (int i = 0; i < m_count; i++) {
		is_included = false;

		for (int j = 0; j < n_count; j++) {
			if (m_array[i] == n_array[j]) {
				cout << "1" << endl;
				is_included = true;
				break;
			}
		}
		
		if (!is_included) cout << "0" << endl;
	}

	return 0;
}
```
naive하게 짠 코드이다. 당연히 시간 초과가 발생한다.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

bool binary_search(int target, vector<long long int> array) {
	int start = 0;
	int end = array.size() - 1;
	int mid;

	while (start <= end) {
		mid = (start + end) / 2;

		if (target == array[mid]) {
			return true;
		}
		else if (target < array[mid]) {
			end = mid - 1;
		}
		else if (target > array[mid]) {
			start = mid + 1;
		}
	}

	return false;
}

int main() {
	ios_base::sync_with_stdio(0);cin.tie(0);

	int n_count, m_count;
	bool is_included = false;

	cin >> n_count;
	vector<long long int> n_array(n_count);
	for (int i = 0; i < n_count; i++) {
		cin >> n_array[i];
	}

	sort(n_array.begin(), n_array.end());

	cin >> m_count;

	for (int i = 0; i < m_count; i++) {
		int m;
		cin >> m;
		if (binary_search(m, n_array)) {
			cout << 1 << '\n';
		}
		else {
			cout << 0 << '\n';
		}
	}

	return 0;
}
```
**Binary Search** 알고리즘을 사용하였는데도 시간 초과가 발생한다.

<br>

## 풀이

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

bool binary_search(int target, vector<long long int> &array) {
	int start = 0;
	int end = array.size() - 1;
	int mid;

	while (start <= end) {
		mid = (start + end) / 2;

		if (target == array[mid]) {
			return true;
		}
		else if (target < array[mid]) {
			end = mid - 1;
		}
		else if (target > array[mid]) {
			start = mid + 1;
		}
	}

	return false;
}

int main() {
	ios_base::sync_with_stdio(0);cin.tie(0);

	int n_count, m_count;
	bool is_included = false;

	cin >> n_count;
	vector<long long int> n_array(n_count);
	for (int i = 0; i < n_count; i++) {
		cin >> n_array[i];
	}

	sort(n_array.begin(), n_array.end());

	cin >> m_count;

	for (int i = 0; i < m_count; i++) {
		int m;
		cin >> m;
		if (binary_search(m, n_array)) {
			cout << 1 << '\n';
		}
		else {
			cout << 0 << '\n';
		}
	}

	return 0;
}
```

함수의 매개변수에서 `vector`형을 받을 때 **reference**를 사용하였다. <br>
**reference**를 사용하지 않고 **call by value**를 하게 되면, `vector`를 복사하는 데에 **O(n)**의 시간 복잡도가 소요된다.
